CREATE OR REPLACE VIEW ASS_VIEW_FUNC_PERMITIDO_RESERV AS
SELECT A.CNOMEFUNCI NOME,
       DECODE(A.CSITUFUNCI, 'A', 'ATIVO', 'INATIVO') SITUACAO,
       A.NCODIFUNCI,
       A.CCRACFUNCI
  FROM ASSFUNCI A
 WHERE A.NEMPRFUNCI IN (1, 2)
   AND SUBSTR(A.CCRACFUNCI, 1, 2) IN ('01', '02', '03', '05', '09');
